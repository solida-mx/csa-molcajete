  if(ms<=0){t.innerHTML='⌛ <b>0:00</b>';t.classList.add("low");expira();return;}

let optState=null;
function openOpts(id){
  if(!chequeaSesion())return;
  const it=ALL[id];
  if(AGOTADOS.has(id)){showToast(it.n+" se agotó por hoy 😔");return;}
  optState={id,q:1,sel:it.ops.map(o=>[o.t,o.c[0][0],o.c[0][1]])};
  $("sheet").innerHTML=`<div class="grip"></div>
    <div class="oname">${it.img} ${it.n}</div>
    ${it.d?`<div class="odesc">${it.d}</div>`:""}
    ${it.ops.map((o,gi)=>`<div class="ogrp"><label>${o.t}</label><div class="ochips" data-g="${gi}">
      ${o.c.map((c,ci)=>`<div class="ochip ${ci===0?'active':''}" data-g="${gi}" data-c="${ci}">${c[0]}${c[1]>0?` <small>+${fmt(c[1])}</small>`:''}</div>`).join("")}
    </div></div>`).join("")}
    <div class="oqty"><div class="qbox"><button id="oMinus">−</button><span class="q" id="oQ">1</span><button id="oPlus">+</button></div></div>
    <button class="oadd" id="oAdd">Agregar a mi orden · ${fmt(it.p)}</button>
    <button class="close" id="oClose">Cancelar</button>`;
  const refresh=()=>{
    const unit=it.p+optState.sel.reduce((a,s)=>a+s[2],0);
    $("oQ").textContent=optState.q;
    $("oAdd").textContent="Agregar a mi orden · "+fmt(unit*optState.q);
  };
  $("sheet").querySelectorAll(".ochip").forEach(ch=>ch.onclick=()=>{
    const gi=+ch.dataset.g,ci=+ch.dataset.c,o=it.ops[gi];
    optState.sel[gi]=[o.t,o.c[ci][0],o.c[ci][1]];
    $("sheet").querySelectorAll(`.ochip[data-g="${gi}"]`).forEach(x=>x.classList.toggle("active",+x.dataset.c===ci));
    refresh();
  });
  $("oMinus").onclick=()=>{optState.q=Math.max(1,optState.q-1);refresh();};
  $("oPlus").onclick=()=>{optState.q++;refresh();};
  $("oAdd").onclick=()=>{
    addLine(optState.id,optState.sel,optState.q);
    renderCtrl(optState.id);syncBar();closeSheet();
    showToast(ALL[optState.id].n+" agregado ✓");
  };
  $("oClose").onclick=closeSheet;
  refresh();showSheet();
}

function destino(){

    return {ok:!!nom,label:"🥡 PARA LLEVAR"+(nom?" — "+nom:"")+(h?" (recoger "+h+")":""),warn:"Escribe un nombre para recoger"};}

function orderText(f){
  const d=destino(),L=[];

  L.push("Folio "+f+"  |  "+new Date().toLocaleTimeString("es-MX",{hour:"2-digit",minute:"2-digit"}));
  L.push("*"+d.label+"*");L.push("");
  CART.forEach(l=>{const it=ALL[l.id];
    L.push(l.q+"×  "+it.n+"   "+fmt(lineUnit(l)*l.q));
    if(l.sel.length)L.push("      ▸ "+selTxt(l));});
  L.push("");L.push("*TOTAL: "+fmt(total())+"*");L.push("Pago: "+pago);
  const notas=$("notas")&&$("notas").value.trim();if(notas)L.push("Notas: "+notas);
  L.push("");L.push("— Pedido enviado desde el menú digital");
  return L.join("\n");
}

async function enviar(){
  if(!chequeaSesion())return;
  if(count()===0){showToast("Tu orden está vacía");return;}
  const ago=CART.filter(l=>AGOTADOS.has(l.id));
  if(ago.length){showToast("Se agotó: "+ago.map(l=>ALL[l.id].n).join(", ")+". Quítalo de tu orden.");return;}
  const d=destino();if(!d.ok){showToast(d.warn);return;}
  reiniciaContador();
  const f=folio();
  if(MODO_COMANDAS!=="api"){enviarWA(f);return;}
  const btn=$("gen");btn.disabled=true;btn.textContent="Enviando…";
  const notas=$("notas")&&$("notas").value.trim();
  const cuerpo={folio:f,tipo:orderType,destino:d.label,pago,notas:notas||"",
    total:total(),items:CART.map(l=>({q:l.q,n:ALL[l.id].n,sel:selTxt(l),imp:lineUnit(l)*l.q}))};
  try{
    const r=await fetch("/api/comandas",{method:"POST",
      headers:{"content-type":"application/json"},body:JSON.stringify(cuerpo)});
    if(!r.ok)throw new Error(r.status);
    const res=await r.json();
    ordenEnviada(res.folio||f,d);
  }catch(e){
    /* Respaldo: si el API no responde (sin señal o sin configurar), va por WhatsApp */
    btn.disabled=false;btn.textContent="Enviar mi orden a cocina";
    enviarWA(f);
  }
}
